def execute(method:, path:, params: {})
  raise NoApiKey if Zype.configuration.api_key.to_s.empty?

  resp = self.send(method, path: path, params: params)
  if resp.success?
    resp['response']
  else
    error!(code: resp.code, message: resp['message'])
  end
end
